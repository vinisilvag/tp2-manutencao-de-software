def _log_timeout_and_verify(method, kwargs):
    """Helper function to log timeout and verify parameters."""
    if "timeout" in kwargs:
        if isinstance(kwargs["timeout"], (int, float)):
            print(f"{method.upper()}: Timeout numérico detectado: {kwargs['timeout']}")
        elif isinstance(kwargs["timeout"], tuple):
            print(f"{method.upper()}: Timeout de conexão/leitura detectado: {kwargs['timeout']}")
    else:
        print(f"{method.upper()}: Nenhum timeout especificado, usando default.")

    if "verify" in kwargs:
        if kwargs["verify"] is False:
            print(f"{method.upper()}: Verificação SSL desabilitada.")
        elif isinstance(kwargs["verify"], str):
            print(f"{method.upper()}: Bundle CA personalizado: {kwargs['verify']}")
    else:
        print(f"{method.upper()}: Verificação SSL habilitada (default).")
